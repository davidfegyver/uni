```
#include <stdio.h>

int main(){
        int num = 0;
        int orig = 0;
        int rev = 0;

        printf("Adj nekem egy számot!");
        scanf("%d", &num);
        orig = num;
        while (num > 0){
                int dig = num % 10;
                rev = rev * 10 + dig;
                num = num / 10;
        }

        printf(orig == rev ? "Meow!\n" : "Not meow!\n");

}
```
